import 'package:flutter/material.dart';
import 'dart:async';
import 'package:firebase_core/firebase_core.dart';
import 'firebase_options.dart'; // auto-generated by FlutterFire CLI or setup assistant
import 'package:firebase_database/firebase_database.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp(options: DefaultFirebaseOptions.currentPlatform);
  runApp(const TimerApp());
}

class TimerApp extends StatelessWidget {
  const TimerApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Smart Timer App',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.deepPurple),
        useMaterial3: true,
      ),
      home: const TimerHomePage(),
    );
  }
}

class TimerHomePage extends StatefulWidget {
  const TimerHomePage({super.key});

  @override
  State<TimerHomePage> createState() => TimerHomePageState();
}

class TimerHomePageState extends State<TimerHomePage> {
  final TextEditingController hoursController = TextEditingController();
  final TextEditingController minutesController = TextEditingController();
  final TextEditingController secondsController = TextEditingController();
  final databaseRef = FirebaseDatabase.instance.ref('Timer_App');

  String displayText = '';
  int totalSeconds = 0;
  int hours = 0;
  int minutes = 0;
  int seconds = 0;
  Timer? countdownTimer;
  String? status;
  int? duration;
  bool showRemainingTime = true;

  @override
  void initState() {
    super.initState();

    databaseRef.onValue.listen((event) {
      final data = event.snapshot.value;
      if (data is Map) {
        setState(() {
          status = data['status'];
          duration = data['duration'];
          totalSeconds = duration ?? 0;
        });
      }

      if (status == 'startWithEncoder') {
        startCountdown(totalSeconds);
        setState(() {
          showRemainingTime = true;
        });
      } else if (status == 'pause') {
        countdownTimer?.cancel();
      } else if (status == 'resume') {
        startCountdown(totalSeconds);
      } else if (status == 'reset') {
        countdownTimer?.cancel();
        setState(() {
          displayText = 'Timer Reset';
          totalSeconds = 0;
        });
      }

      totalSeconds = duration ?? 0;
    });
  }

  Future<void> setTimer() async {
    hours = int.tryParse(hoursController.text) ?? 0;
    minutes = int.tryParse(minutesController.text) ?? 0;
    seconds = int.tryParse(secondsController.text) ?? 0;

    if (hours > 23 || minutes > 59 || seconds > 59) {
      setState(() {
        displayText = 'Please enter a valid time. (Max : 24h 59m 59s)';
      });
      return;
    }

    totalSeconds = hours * 3600 + minutes * 60 + seconds;

    final DatabaseReference dbRef = FirebaseDatabase.instance.ref("Timer_App");

    await dbRef.set({'duration': totalSeconds, 'status': 'startWithApp'});

    startCountdown(totalSeconds);
  }

  void formatTimer(int totalSeconds) {
    if (!showRemainingTime) return;
    hours = totalSeconds ~/ 3600;
    minutes = (totalSeconds % 3600) ~/ 60;
    seconds = (totalSeconds % 60);

    setState(() {
      displayText = 'Plug will run for ${hours}h ${minutes}m ${seconds}s.';
    });
  }

  void startCountdown(int totalSeconds) {
    this.totalSeconds = totalSeconds;
    countdownTimer?.cancel();
    showRemainingTime = true;

    countdownTimer = Timer.periodic(Duration(seconds: 1), (timer) {
      if (this.totalSeconds > 0) {
        setState(() {
          this.totalSeconds--;
          formatTimer(this.totalSeconds);
        });
      } else {
        timer.cancel();
        setState(() {
          showRemainingTime = false;
          displayText = 'Countdown completed.';
        });
      }
    });
  }

  Future<void> resetTimer() async {
    setState(() {
      showRemainingTime = false;
      displayText = 'Timer Reset';
      hoursController.clear();
      minutesController.clear();
      secondsController.clear();
      totalSeconds = 0;
    });
    final DatabaseReference dbRef = FirebaseDatabase.instance.ref("Timer_App");

    await dbRef.set({'duration': totalSeconds, 'status': 'reset'});
  }

  Future<void> pauseTimer() async {
    countdownTimer?.cancel();
    final DatabaseReference dbRef = FirebaseDatabase.instance.ref("Timer_App");

    await dbRef.set({'duration': totalSeconds, 'status': 'pause'});
  }

  Future<void> resumeTimer() async {
    startCountdown(totalSeconds);
    final DatabaseReference dbRef = FirebaseDatabase.instance.ref("Timer_App");
    await dbRef.set({'duration': totalSeconds, 'status': 'resume'});
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Smart Plug Timer')),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Column(
          children: [
            const Text('Set timer duration:'),
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: [
                Flexible(
                  child: TextField(
                    controller: hoursController,
                    keyboardType: TextInputType.number,
                    decoration: const InputDecoration(labelText: 'Hours'),
                  ),
                ),
                Flexible(
                  child: TextField(
                    controller: minutesController,
                    keyboardType: TextInputType.number,
                    decoration: const InputDecoration(labelText: 'Minutes'),
                  ),
                ),
                Flexible(
                  child: TextField(
                    controller: secondsController,
                    keyboardType: TextInputType.number,
                    decoration: const InputDecoration(labelText: 'Seconds'),
                  ),
                ),
              ],
            ),
            const SizedBox(height: 20),
            ElevatedButton(onPressed: setTimer, child: const Text('Set Timer')),
            const SizedBox(height: 20),
            ElevatedButton(
              onPressed: resetTimer,
              child: const Text('Reset Timer'),
            ),
            const SizedBox(height: 20),
            ElevatedButton(
              onPressed: pauseTimer,
              child: const Text('Pause Timer'),
            ),
            const SizedBox(height: 20),
            ElevatedButton(
              onPressed: resumeTimer,
              child: const Text('Resume Timer'),
            ),
            const SizedBox(height: 20),
            Text(displayText, style: const TextStyle(fontSize: 18)),
          ],
        ),
      ),
    );
  }
}
